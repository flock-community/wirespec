import enum
from ..wirespec import T, Wirespec, _raise
@dataclass
class DutchPostalCode(Wirespec.Refined[str]):
    value: str
    def validate(self) -> bool:
        return bool(re.match(r"/^([0-9]{4}[A-Z]{2})$/g", self.value))
    def __str__(self) -> str:
        return self.value
